# AI Features Frontend Implementation Plan

## Overview
**Feature:** AI-Powered Question & Answer Generation UI  
**Goal:** Build user interfaces for AI-generated content display, lie generation, and configuration with a modern, minimal, yet playful design.

### Scope
- AI configuration UI (admin/host only)
- Question display with AI indicator
- Big Fish answer reveal (AI-generated)
- Red Herring lie generation interface
- "Generate More Lies" button with loading states
- Error handling with fallback word bank notification
- Category/difficulty selection (optional)

### useAIGeneration (`hooks/useAIGeneration.ts`)
```typescript
'use client';

import { useState, useCallback } from 'react';
import apiClient from '@/lib/axios';

interface AIConfig {
  apiKey: string;
  model: string;
  baseURL: string;
}

interface GenerationResult {
  question: string;
  correctAnswer: string;
  bluffSuggestions: string[];
  model: string;
}

export const useAIGeneration = (roomCode: string) => {
  const [isGenerating, setIsGenerating] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const [result, setResult] = useState<GenerationResult | null>(null);

  const generateRound = useCallback(async (
    category?: string,
    difficulty?: 'easy' | 'medium' | 'hard'
  ) => {
    setIsGenerating(true);
    setError(null);

    try {
      const response = await apiClient.post(
        `/api/rooms/${roomCode}/generate-round`,
        { category, difficulty, regenerate: true }
      );

      setResult(response.data);
      return response.data;
    } catch (err: any) {
      setError(err.response?.data?.error || err.message || 'Generation failed');
      return null;
    } finally {
      setIsGenerating(false);
    }
  }, [roomCode]);

  const generateLie = useCallback(async (
    question: string,
    correctAnswer: string,
    existingAnswers: string[]
  ): Promise<string | null> => {
    setIsGenerating(true);
    setError(null);

    try {
      const response = await apiClient.post(
        `/api/rooms/${roomCode}/generate-lie`,
        { question, correctAnswer, existingAnswers }
      );

      return response.data.data.lieSuggestion;
    } catch (err: any) {
      setError(err.response?.data?.error || err.message || 'Lie generation failed');
      return null;
    } finally {
      setIsGenerating(false);
    }
  }, [roomCode]);

  return {
    isGenerating,
    error,
    result,
    generateRound,
    generateLie,
  };
};
```

### useAIConfig (`hooks/useAIConfig.ts`)
```typescript
'use client';

import { useState, useEffect, useCallback } from 'react';
import apiClient from '@/lib/axios';

interface AIConfig {
  apiKey: string;
  model: string;
  baseURL: string;
}

const DEFAULT_CONFIG: AIConfig = {
  apiKey: '',
  model: 'gpt-3.5-turbo',
  baseURL: 'https://api.openai.com/v1',
};

export const useAIConfig = (roomCode: string) => {
  const [config, setConfig] = useState<AIConfig>(DEFAULT_CONFIG);
  const [isLoading, setIsLoading] = useState(true);
  const [isSaving, setIsSaving] = useState(false);

  // Load config on mount
  useEffect(() => {
    const loadConfig = async () => {
      try {
        const response = await apiClient.get(`/api/rooms/${roomCode}/ai-config`);
        if (response.data.success) {
          setConfig(response.data.data);
        }
      } catch (err) {
        console.error('Failed to load AI config:', err);
      } finally {
        setIsLoading(false);
      }
    };

    if (roomCode) {
      loadConfig();
    }
  }, [roomCode]);

  const saveConfig = useCallback(async (newConfig: Partial<AIConfig>) => {
    setIsSaving(true);

    try {
      const response = await apiClient.put(
        `/api/rooms/${roomCode}/ai-config`,
        { ...config, ...newConfig }
      );

      if (response.data.success) {
        setConfig(response.data.data);
      }
    } catch (err) {
      console.error('Failed to save AI config:', err);
    } finally {
      setIsSaving(false);
    }
  }, [roomCode, config]);

  const testConnection = useCallback(async (): Promise<boolean> => {
    try {
      const response = await apiClient.post(
        `/api/rooms/${roomCode}/ai-test`,
        config
      );
      return response.data.success;
    } catch (err) {
      console.error('Connection test failed:', err);
      return false;
    }
  }, [roomCode, config]);

  return {
    config,
    isLoading,
    isSaving,
    saveConfig,
    testConnection,
  };
};
```
